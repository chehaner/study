#### 大数相加
```javascript
function sum(a, b){
    const len = Math.max(a.length, b.length);
    a = a.padStart(len, '0');
    b = b.padStart(len, '0');
    let carry = 0;
    let result = '';
    for(let i=len-1; i>=0; i--){
        let sum = parseInt(a[i]) + parseInt(b[i]) + carry;
        result = (sum%10) + result;
        carry = Math.floor(sum/10);
    }
    if (carry > 0) {
        result = carry + result;
    }
    return result;
}
```
